233  template<typename Command>
234  void executeCommand(Command command)
235  {
236  if constexpr (std::is_same_v<Command, XLCommandSetSheetName>) {
237  m_appProperties.setSheetName(command.sheetName(), command.newName());
238  m_workbook.setSheetName(command.sheetID(), command.sheetName());
239  }
240 
241  else if constexpr (std::is_same_v<Command, XLCommandSetSheetVisibility>) { // NOLINT
242  }
243 
244  else if constexpr (std::is_same_v<Command, XLCommandSetSheetColor>) {
245  }
246 
247  else if constexpr (std::is_same_v<Command, XLCommandSetSheetIndex>) {
248  auto sheetName = executeQuery(XLQuerySheetName(command.sheetID())).sheetName();
249  m_workbook.setSheetIndex(sheetName, command.sheetIndex());
250  }
251 
252  else if constexpr (std::is_same_v<Command, XLCommandAddSharedStrings>) {
253  std::string sharedStrings {
254  "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n"
255  "<sst xmlns=\"http://schemas.openxmlformats.org/spreadsheetml/2006/main\" count=\"1\" uniqueCount=\"1\">\n"
256  " <si>\n"
257  " <t/>\n"
258  " </si>\n"
259  "</sst>"
260  };
261  m_contentTypes.addOverride("/xl/sharedStrings.xml", XLContentType::SharedStrings);
262  m_wbkRelationships.addRelationship(XLRelationshipType::SharedStrings, "sharedStrings.xml");
263  m_archive.addEntry("xl/sharedStrings.xml", sharedStrings);
264  }
265 
266  else if constexpr (std::is_same_v<Command, XLCommandAddWorksheet>) {
267  std::string emptyWorksheet {
268  "<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"yes\"?>\n"
269  "<worksheet xmlns=\"http://schemas.openxmlformats.org/spreadsheetml/2006/main\""
270  " xmlns:r=\"http://schemas.openxmlformats.org/officeDocument/2006/relationships\""
271  " xmlns:mc=\"http://schemas.openxmlformats.org/markup-compatibility/2006\" mc:Ignorable=\"x14ac\""
272  " xmlns:x14ac=\"http://schemas.microsoft.com/office/spreadsheetml/2009/9/ac\">"
273  "<dimension ref=\"A1\"/>"
274  "<sheetViews>"
275  "<sheetView workbookViewId=\"0\"/>"
276  "</sheetViews>"
277  "<sheetFormatPr baseColWidth=\"10\" defaultRowHeight=\"16\" x14ac:dyDescent=\"0.2\"/>"
278  "<sheetData/>"
279  "<pageMargins left=\"0.7\" right=\"0.7\" top=\"0.75\" bottom=\"0.75\" header=\"0.3\" footer=\"0.3\"/>"
280  "</worksheet>"
281  };
282  m_contentTypes.addOverride(command.sheetPath(), XLContentType::Worksheet);
283  m_wbkRelationships.addRelationship(XLRelationshipType::Worksheet, command.sheetPath().substr(4));
284  m_appProperties.appendSheetName(command.sheetName());
285  m_archive.addEntry(command.sheetPath().substr(1), emptyWorksheet);
286  m_data.emplace_back(
287  /* parentDoc */ this,
288  /* xmlPath */ command.sheetPath().substr(1),
289  /* xmlID */ m_wbkRelationships.relationshipByTarget(command.sheetPath().substr(4)).id(),
290  /* xmlType */ XLContentType::Worksheet);
291  }
292 
293  else if constexpr (std::is_same_v<Command, XLCommandAddChartsheet>) {
294  }
295 
296  else if constexpr (std::is_same_v<Command, XLCommandDeleteSheet>) {
297  m_appProperties.deleteSheetName(command.sheetName());
298  auto sheetPath = "/xl/" + m_wbkRelationships.relationshipById(command.sheetID()).target();
299  m_archive.deleteEntry(sheetPath.substr(1));
300  m_contentTypes.deleteOverride(sheetPath);
301  m_wbkRelationships.deleteRelationship(command.sheetID());
302  m_data.erase(std::find_if(m_data.begin(), m_data.end(), [&](const XLXmlData& item) {
303  return item.getXmlPath() == sheetPath.substr(1);
304  }));
305  }
306 
307  else if constexpr (std::is_same_v<Command, XLCommandCloneSheet>) {
308  auto internalID = m_workbook.createInternalSheetID();
309  auto sheetPath = "/xl/worksheets/sheet" + std::to_string(internalID) + ".xml";
310  if (m_workbook.sheetExists(command.cloneName()))
311  throw XLException("Sheet named \"" + command.cloneName() + "\" already exists.");
312 
313  if (m_wbkRelationships.relationshipById(command.sheetID()).type() == XLRelationshipType::Worksheet) {
314  m_contentTypes.addOverride(sheetPath, XLContentType::Worksheet);
315  m_wbkRelationships.addRelationship(XLRelationshipType::Worksheet, sheetPath.substr(4));
316  m_appProperties.appendSheetName(command.cloneName());
317  m_archive.addEntry(sheetPath.substr(1), std::find_if(m_data.begin(), m_data.end(), [&](const XLXmlData& data) {
318  return data.getXmlPath().substr(3) ==
319  m_wbkRelationships.relationshipById(command.sheetID()).target();
320  })->getRawData());
321  m_data.emplace_back(
322  /* parentDoc */ this,
323  /* xmlPath */ sheetPath.substr(1),
324  /* xmlID */ m_wbkRelationships.relationshipByTarget(sheetPath.substr(4)).id(),
325  /* xmlType */ XLContentType::Worksheet);
326  }
327  else {
328  m_contentTypes.addOverride(sheetPath, XLContentType::Chartsheet);
329  m_wbkRelationships.addRelationship(XLRelationshipType::Chartsheet, sheetPath.substr(4));
330  m_appProperties.appendSheetName(command.cloneName());
331  m_archive.addEntry(sheetPath.substr(1), std::find_if(m_data.begin(), m_data.end(), [&](const XLXmlData& data) {
332  return data.getXmlPath().substr(3) ==
333  m_wbkRelationships.relationshipById(command.sheetID()).target();
334  })->getRawData());
335  m_data.emplace_back(
336  /* parentDoc */ this,
337  /* xmlPath */ sheetPath.substr(1),
338  /* xmlID */ m_wbkRelationships.relationshipByTarget(sheetPath.substr(4)).id(),
339  /* xmlType */ XLContentType::Chartsheet);
340  }
341 
342  m_workbook.prepareSheetMetadata(command.cloneName(), internalID);
343  }
344 
345  else {
346  throw XLException("Invalid command object.");
347  }

356  template<typename Query>
357  Query executeQuery(Query query) const
358  {
359  if constexpr (std::is_same_v<Query, XLQuerySheetName>) { // NOLINT
360  query.setSheetName(m_workbook.sheetName(query.sheetID()));
361  return query;
362  }
363 
364  else if constexpr (std::is_same_v<Query, XLQuerySheetIndex>) {
365  return query;
366  }
367 
368  else if constexpr (std::is_same_v<Query, XLQuerySheetVisibility>) {
369  query.setSheetVisibility(m_workbook.sheetVisibility(query.sheetID()));
370  return query;
371  }
372 
373  else if constexpr (std::is_same_v<Query, XLQuerySheetType>) {
374  if (m_wbkRelationships.relationshipById(query.sheetID()).type() == XLRelationshipType::Worksheet)
375  query.setSheetType(XLContentType::Worksheet);
376  else
377  query.setSheetType(XLContentType::Chartsheet);
378 
379  return query;
380  }
381 
382  else if constexpr (std::is_same_v<Query, XLQuerySheetID>) {
383  query.setSheetVisibility(m_workbook.sheetVisibility(query.sheetName()));
384  return query;
385  }
386 
387  else if constexpr (std::is_same_v<Query, XLQuerySheetRelsID>) {
388  query.setSheetID(m_wbkRelationships.relationshipByTarget(query.sheetPath().substr(4)).id());
389  return query;
390  }
391 
392  else if constexpr (std::is_same_v<Query, XLQuerySheetRelsTarget>) {
393  query.setSheetTarget(m_wbkRelationships.relationshipById(query.sheetID()).target());
394  return query;
395  }
396 
397  else {
398  throw XLException("Invalid query object.");
399  }

438 
439  else if constexpr (std::is_same_v<Query, XLQuerySharedStrings>) {
440  query.setSharedStrings(&m_sharedStrings);
441  return query;
442  }
443 
444  else if constexpr (std::is_same_v<Query, XLQueryXmlData>) {
445  auto result =
446  std::find_if(m_data.begin(), m_data.end(), [&](const XLXmlData& item) { return item.getXmlPath() == query.xmlPath(); });
447  if (result == m_data.end()) throw XLException("Path does not exist in zip archive.");
448  query.setXmlData(&*result);
449  return query;
450  }
